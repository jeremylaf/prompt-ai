  <script>

    function genererPrompt() {

      const personnage = document.getElementById('personnage').value;

      const action = document.getElementById('action').value;

      const lieu = document.getElementById('lieu').value;

      const moment = document.getElementById('moment').value;

      const ambiance = document.getElementById('ambiance').value;

      const style = document.getElementById('style').value;

      const camera = document.getElementById('camera').value;

      const duree = document.getElementById('duree').value;


      const prompt = `Vidéo ${duree ? `de ${duree} ` : ''}montrant ${personnage} qui ${action}, dans ${lieu}, ${moment}. ` +

                      `Ambiance ${ambiance}, style visuel ${style}, caméra : ${camera}.`;


      document.getElementById('output').textContent = prompt;

    }
